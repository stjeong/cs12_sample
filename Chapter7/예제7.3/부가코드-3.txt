
/* ================= 7.2 ?? 연산자 ================= */

class Program
{
    static void Main(string[] args)
    {
        string txt = null;

        if (txt == null)
        {
            Console.WriteLine("(null)");
        }
        else
        {
            Console.WriteLine(txt);
        }

        Console.WriteLine(txt ?? "(null)");
    }
}
